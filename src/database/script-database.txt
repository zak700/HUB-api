create table escolaridades (
	id_escolaridade SERIAL PRIMARY KEY NOT NULL,
	escolaridade_nome  VARCHAR (255),
)







create table admins (
	id_admin  INTEGER PRIMARY KEY AUTOINCREMENT,
    name      VARCHAR (50),
	email     VARCHAR (100),
	password  VARCHAR (100)
);

create table doctors (
	id_doctor SERIAL PRIMARY KEY NOT NULL,
	name      VARCHAR (50),
	specialty VARCHAR (50),
	icon      VARCHAR (10)
);

create table services (
	id_service  SERIAL PRIMARY KEY NOT NULL,
	description VARCHAR (50)
);

create table doctors_services (
	id_doctor_service SERIAL PRIMARY KEY NOT NULL,
	id_doctor  INTEGER,
	id_service INTEGER,
	price      DECIMAL(9,2),
	foreign key (id_doctor) references doctors(id_doctor),
	foreign key (id_service) references services(id_service)
);

create table appointments (
	id_appointment SERIAL PRIMARY KEY NOT NULL,
	id_doctor    INTEGER,
	id_service   INTEGER,
	id_user      INTEGER,
	booking_date DATE,
	booking_hour VARCHAR(5),
	
	foreign key (id_doctor) references doctors(id_doctor),
    foreign key (id_service) references services(id_service),
    foreign key (id_user) references users(id_user)
);
